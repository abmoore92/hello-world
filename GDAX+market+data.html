def convertToNumeric(df,cols):
    df2 = df.copy()
    for c in cols:
        df2[c] = pd.to_numeric(df2[c])
    return df2

# Get lots of trades
# side means the maker side of the trade (liquidity maker, the order that was sitting on the book, 
#                                          - while the taker is the order that comes in and accepts that deal)
# implications: buy means price decreased (maker was sitting on lower price), sell means price increased (maker was sitting on higher price)
trades =  requests.get('https://api.gdax.com/products/%s/trades'%product)
tradeData = pd.DataFrame.from_dict(trades.json())
tradeData['time']=pd.to_datetime(tradeData['time'])
tradeData['price']=pd.to_numeric(tradeData['price'])
tradeData['size']=pd.to_numeric(tradeData['size'])
#tradeData.plot(x='time', y='price', style='o')
tradeData['sellprice']=tradeData.loc[tradeData['side']=='sell']['price']
tradeData['buyprice']=tradeData.loc[tradeData['side']=='buy']['price']

tradeData.plot(x='time',y=['sellprice','buyprice'],style='o')
